// Return the n-th number in the Fibonacci Sequence

function nthFibo(n) { 
    if ( n<=0) {
        return undefined
    };
    let fibo = [0, 1];
    for (let i=2; i<n; i++) {
        fibo[i] = fibo[i-2] + fibo[i-1];
    }
    return fibo[n-1];
}
